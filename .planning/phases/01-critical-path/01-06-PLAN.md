<!-- tel: URI spec -->
Format: tel:+33612345678
- Works on iOS (Siri phone app) and Android as a one-tap call intent
- Must use the E.164 format phone number (already in this format from Google Ads)
- SMS body: "Lead: Marie D.\n\"Fuite d eau...\"\nAppeler: tel:+33612345678"

<!-- Twilio owner SMS — same HTTP Request pattern as prospect -->

To   → {{ $env.OWNER_PHONE }}      (owner's mobile E.164 — from env var, not config file)
From → {{ $env.TWILIO_SENDER_ID }}
Body → {{ $json.owner_sms }}

"parameters": {
  "jsCode": "// NOTIF-01, NOTIF-02: Owner notification with tel: one-tap callback link\nconst name = $json.name || 'Prospect';\nconst request = $json.request || '';\nconst phone = $json.phone || null;\nconst email = $json.email || 'non fourni';\nconst claudeMsg = $json.claude_message || '';\n\n// Build owner SMS — keep under 320 chars (2 segments)\nlet ownerSms;\nif (phone) {\n  // Phone available: include one-tap tel: link\n  ownerSms = `Lead: ${name}\\n\"${request.substring(0, 60)}\"\\nAppeler: tel:${phone}`;\n} else {\n  // Email only: no tel: link possible\n  ownerSms = `Lead: ${name}\\n\"${request.substring(0, 60)}\"\\nEmail: ${email}\\n(Pas de tél)`;\n}\n\n// Truncate to 320 chars max\nownerSms = ownerSms.substring(0, 320);\n\n// Carry all fields forward for potential future nodes\nreturn [{\n  json: {\n    lead_id: $json.lead_id,\n    name,\n    phone,\n    email,\n    request,\n    claude_message: claudeMsg,\n    owner_sms: ownerSms\n  }\n}];"
}
